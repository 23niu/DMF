191  protected function _get_sql_from_condition($condition = [])
192  {
193  if (count($condition) <= 0) {
194  return ['query' => '', 'params' => []];
195  }
196  $result = [
197  'queries' => [],
198  'params' => []
199  ];
200  $i = 0;
201  foreach ($condition as $field_cond => $value) {
202  $data = explode('__', $field_cond);
203  $field_name = $data[0];
205  if ($i == 0) {
206  $precond = '';
207  }
208  elseif (substr($data[0], 0, 1) == '~') {
209  $precond = ' OR ';
210  $field_name = substr($data[0], 1);
211  }
212  else {
213  $precond = ' AND ';
214  }
216  if (count($data) == 1) {
217  $result['queries'][] = $precond . $field_name . '=:' . $field_name;
218  $result['params'][$field_name] = $value;
219  }
220  else {
221  $cond = $data[1];
222  switch ($cond) {
224  case 'equal':
225  $result['queries'][] = $precond . $field_name . '=:' . $field_name;
226  $result['params'][$field_name] = $value;
227  break;
229  case 'not_equal':
230  $result['queries'][] = $precond . $field_name . '!=:' . $field_name;
231  $result['params'][$field_name] = $value;
232  break;
234  case 'gt':
235  $result['queries'][] = $precond . $field_name . '>:' . $field_name;
236  $result['params'][$field_name] = $value;
237  break;
239  case 'gte':
240  $result['queries'][] = $precond . $field_name . '>=:' . $field_name;
241  $result['params'][$field_name] = $value;
242  break;
244  case 'lt':
245  $result['queries'][] = $precond . $field_name . '<:' . $field_name;
246  $result['params'][$field_name] = $value;
247  break;
249  case 'lte':
250  $result['queries'][] = $precond . $field_name . '<=:' . $field_name;
251  $result['params'][$field_name] = $value;
252  break;
254  case 'startswith':
255  $result['queries'][] = $precond . $field_name . ' LIKE :' . $field_name;
256  $result['params'][$field_name] = $value . '%';
257  break;
259  case 'endswith':
260  $result['queries'][] = $precond . $field_name . ' LIKE :' . $field_name;
261  $result['params'][$field_name] = '%' . $value;
262  break;
264  case 'contains':
265  $result['queries'][] = $precond . $field_name . ' LIKE :' . $field_name;
266  $result['params'][$field_name] = '%' . $value . '%';
267  break;
269  case 'in':
270  if (is_array($value)) {
271  $value = implode(', ', $value);
272  }
273  $result['queries'][] = $precond . $field_name . ' IN(:' . $field_name . ')';
274  $result['params'][$field_name] = implode(', ', $value);
275  break;
277  case 'is_null':
278  if ($value === true) {
279  $result['queries'][] = $precond . $field_name . ' IS NULL ';
280  }
281  else {
282  $result['queries'][] = $precond . $field_name . ' IS NOT NULL ';
283  }
284  break;
285  default:
286  $result['queries'][] = $precond . $field_name . '=:' . $field_name;
287  $result['params'][$field_name] = $value;
288  break;
289  }
290  }
291  $i++;
292  }
293  return [
294  'query' => ' WHERE ' . implode('', $result['queries']),
295  'params' => $result['params']
296  ];
297  }
