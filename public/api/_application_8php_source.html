83  protected function parse_route()
84  {
85  // Проверяем не был ли текущий объект маршрута уже закеширован ранее
86  if (is_null($this->route_object)) {
87  // Текущая строка запроса
88  $uri = $this->request->request_uri();
89  // Список зарегистрированных маршрутов
90  $routes = Router::data();
91  // Поиск совпадающих со строкой запроса маршрутов
92  foreach ($routes as $regexp => $pattern) {
93  // Заменяем псевдонимы для regexp шаблона
94  $regexp = str_replace(
95  ['@int', '@str', '@alphanum', '@all'],
96  ['[\d]+', '[\w]+', '[\w\d]+', '[\w\d\.\,\-\_\+]+'],
97  $regexp
98  );
99  // Проверяем соответствие текущей строке запроса и паттерну маршрута
100  if (preg_match('~^' . $regexp . '$~i', $uri, $arguments)) {
101  $route_object = [
102  'url' => $this->request->request_uri(),
103  'pattern' => $pattern,
104  'arguments' => array_slice($arguments, 1)
105  ];
106  $this->route_object = $route_object;
107  return $route_object;
108  }
109  }
110  throw new Http404('Данная страница отсутствует на сайте!');
111  }
112  else {
113  return $this->route_object;
114  }
115  }
